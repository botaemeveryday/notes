{{/* Open Graph + Twitter Card meta tags. Picks the share image: - single page → /covers/
/.png - section page → /covers/
/index.png - fallback → /images/avatar.jpg An explicit `cover` param in the page frontmatter overrides everything. */}} {{- $cover := "" -}} {{- $section := .Section -}} {{- with .Params.cover -}} {{- $cover = . -}} {{- else -}} {{- if and .IsPage $section -}} {{- $cover = printf "/covers/%s/%s.png" $section .File.ContentBaseName -}} {{- else if and .IsSection $section -}} {{- $cover = printf "/covers/%s/index.png" $section -}} {{- else -}} {{- $cover = "/images/avatar.jpg" -}} {{- end -}} {{- end -}} {{- $absCover := $cover | absURL -}} {{- $title := .Title | default site.Title -}} {{- $desc := or .Description .Summary site.Params.description -}} {{- $desc = $desc | plainify -}} {{ with $desc }}{{ end }} {{- if .IsPage }}{{ with .Lastmod }} {{ end }}{{ end }} {{ with $desc }}{{ end }}
