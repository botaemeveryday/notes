{{/* Open Graph + Twitter Card meta tags. Picks the share image: - single page → /covers//.png - section page → /covers//index.png - fallback → /images/avatar.jpg Content layout assumption: content/posts///index.md ← lecture content/posts//_index.md ← course landing So the "section" we care about for the cover path is the SECOND path segment ("operation-systems"), not the first ("posts"). An explicit `cover` param in the page frontmatter overrides everything. */}} {{- $cover := "" -}} {{- /* Extract the course slug from the file path. We split File.Dir on "/" and take index 1 (after "posts"). Works for: posts/operation-systems/lecture-10/ → "operation-systems" posts/operation-systems/ → "operation-systems" Falls back to .Section for pages outside posts/. */ -}} {{- $course := "" -}} {{- with .File }} {{- $parts := split (trim .Dir "/") "/" -}} {{- if and (ge (len $parts) 2) (eq (index $parts 0) "posts") -}} {{- $course = index $parts 1 -}} {{- end -}} {{- end -}} {{- with .Params.cover -}} {{- $cover = . -}} {{- else -}} {{- if and .IsPage $course -}} {{- $cover = printf "covers/%s/%s.png" $course .File.ContentBaseName -}} {{- else if and .IsSection $course -}} {{- $cover = printf "covers/%s/index.png" $course -}} {{- else -}} {{- $cover = "images/avatar.jpg" -}} {{- end -}} {{- end -}} {{- $absCover := $cover | absURL -}} {{- $title := .Title | default site.Title -}} {{- $desc := or .Description .Summary site.Params.description -}} {{- $desc = $desc | plainify -}} {{ with $desc }}{{ end }} {{- if .IsPage }}{{ with .Lastmod }} {{ end }}{{ end }} {{ with $desc }}{{ end }}
